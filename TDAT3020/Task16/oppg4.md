Bruk Cæsarchifferet y = x + 3 (mod 2^4) og finn CBC-MACen til de to meldingene

x = 1101 1111 1010 0001
    
    1101 = 13, 13 + 3 = 16, 16 mod 16 = 0000

    1111 XOR 0000 = 1111 = 15, 15 + 3, = 18, 18 mod 16 = 0010

    1010 XOR 0010 = 1000 = 8, 8 + 3 = 11, 11 mod 16 = 1011
    
    0001 XOR 1011 = 1010 = 10, 10 + 3 = 13, 13 mod 16 = 1101

    MAC = 1101
    

x′ = 0010 1100 0001 1111

    0010 = 2, 2 + 3 = 5, 5 mod 16 = 0101

    1100 XOR 0101 = 1001 = 9, 9 + 3 = 12, 12 mod 16 = 1100

    0001 XOR 1100 = 1101 = 13, 13 + 3 = 16, 16 mod 16 = 0000

    1111 XOR 0000 = 1111 = 15, 15 + 3 = 18, 18 mod 16 = 0010

    MAC = 0010
